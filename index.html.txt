    <div class="box">
        <div class="label">CyberSich Wallet</div>
        <div class="value" id="wallet">0.0 $SICH</div>
        
        <div class="label">Mining Progress (0.1 SICH)</div>
        <div class="progress-bar"><div class="fill" id="fill"></div></div>
        
        <button id="btn" onclick="claim()" disabled>WAITING FOR 0.1 SICH...</button>
    </div>

    <script>
        let wallet = 0.0;
        let progress = 0;
        const btn = document.getElementById('btn');
        const fill = document.getElementById('fill');

        function startMining() {
            setInterval(() => {
                if (progress < 100) {
                    progress += 1; // 1% кожну секунду (разом 100 секунд до 0.1 SICH)
                    fill.style.width = progress + "%";
                    if (progress >= 100) {
                        btn.disabled = false;
                        btn.innerText = "CLAIM 0.1 SICH";
                    }
                }
            }, 1000); 
        }

        function claim() {
            wallet = (parseFloat(wallet) + 0.1).toFixed(1);
            document.getElementById('wallet').innerText = wallet + " $SICH";
            progress = 0;
            fill.style.width = "0%";
            btn.disabled = true;
            btn.innerText = "WAITING FOR 0.1 SICH...";
        }

        startMining();
    </script>
